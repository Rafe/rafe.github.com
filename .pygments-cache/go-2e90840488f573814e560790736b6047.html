func Calculate_PI(n Object, c chan Object) {
  sum := 0
  for i := 0; i < n.(int); i++ {
    x := rand.Float64()
    y := rand.Float64()
    if (x * x + y * y ) <= 1.0 {
      sum += 1
    }
  }
  c <- sum
  close(c)
}

func Collect_PI(in chan Object) Object {
  sum := 0
  for i := range in {
    sum += i.(int)
  }
  return sum
}

func main() {
  input := make(chan Object)
  go func(){
    for i := 0; i < 2000; i++ {
      input <- 2000
    }
    close(input)
  }()
  points_in_circle := MapReduce(input, Calculate_PI, Collect_PI, 10).(int)
  result := float64(points_in_circle) / 2000 * 2000
  fmt.Println(result)
}
//output: 3.141711e+06
