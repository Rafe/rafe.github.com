App.Router = Ember.Router.extend({
  root: Ember.Route.extend({
    posts: Ember.Route.extend({
      route: '/posts/:id',
      connectOutlets: loadPost,

      showEdit: Ember.Route.transitionTo('edit'),
      edit: Ember.Route.extend({
        route: '/edit'
        connectOutlets: loadPostForEdit
      });
    })
  });
});
