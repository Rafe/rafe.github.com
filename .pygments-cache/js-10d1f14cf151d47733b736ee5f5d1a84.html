var User = Ember.Object.extend({
      firstName: null,
      lastName: null
    });

    User.reopen({
      name: function(){
        return this.firstName + " " + this.lastName;
      }.property('firstName','lastName')
    });

    var user = User.create({ firstName: "John", lastName: "Doe"});

    console.log(user.get('name')); //John Doe
